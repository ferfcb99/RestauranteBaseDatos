CREATE DATABASE IF NOT EXISTS RestauranteScrum;
use RestauranteScrum;

create table if not exists Restaurante(
	id int not null auto_increment primary key,
	nombre varchar(100) not null unique,
	tipo varchar(50) 
);

INSERT INTO Restaurante(nombre, tipo) VALUES('Restaurante Vegano', 'Comida vegana'), 
											('Comida rapida', 'Alimentos altos en grasas'),
                                            ('Restaurante neutro', 'Alimentos reguladores'),
                                            ('Restaurante 4', 'Cereales');
                                            
                                            select * from Restaurante;

create table if not exists Tipo(
	id int not null auto_increment primary key,
    nombre varchar(100) not null unique
);

INSERT INTO Tipo(nombre) Values('Vegano'), ('Vegetariano'), ('Legumbres'), ('Regulares');

select * from Tipo;

create table if not exists Platillo(
	id int not null auto_increment primary key,
    nombre varchar(100) not null unique,
    tamano varchar(100),
	precio numeric(5,2) not null,
    restaurante_id int not null, 
    tipo_id int not null,
    foreign key(restaurante_id) references Restaurante(id) ON DELETE CASCADE ON UPDATE CASCADE,
    foreign key(tipo_id) references Tipo(id) ON DELETE CASCADE ON UPDATE CASCADE
);
